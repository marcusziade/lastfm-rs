Quick Start

Install CLI

# Install from source
cargo install --path . --bin lastfm-cli

# Or use the install script
./scripts/install-cli.sh

Authenticate

# Login to Last.fm
lastfm-cli auth login

# Check status
lastfm-cli auth status

Examples

# Get your recent tracks
lastfm-cli my recent-tracks --limit 10

# See your top artists this month
lastfm-cli my top-artists --period 1month

# Export your library
lastfm-cli my top-tracks --period overall --limit 1000 -o json > my_library.json
# Find similar artists
lastfm-cli artist similar "Radiohead" --limit 10

# Discover music by country
lastfm-cli geo top-tracks "Japan" --limit 20

# Explore genres
lastfm-cli tag top-artists "shoegaze" -o table
# Compare music taste with friends
lastfm-cli user compare "YourUsername" "FriendUsername"

# Track listening habits over time
lastfm-cli my weekly-chart-list | \
  lastfm-cli my weekly-track-chart --from $(date -d '1 month ago' +%s)

# Export to CSV for analysis
lastfm-cli my top-tracks --period overall -o json | \
  jq -r '.toptracks.track[] | [.name, .artist.name, .playcount] | @csv' > tracks.csv
